- 少直接使用CLASS选择器

- 多用父子关系, 少用嵌套关系

### 使用ready事件

```javascript
// 只会触发第2次
window.onload = function(){
    alert("一次调用load事件");
}
window.onload = function(){
    alert("二次调用load事件");
}


// 2次都会调用

     (function(){
        alert("一次调用load事件");
    })();
     (function(){
        alert("二次调用load事件");
    })()

    alert("一次调用load事件")
});
$(function(){
    alert("二次调用load事件")
});

// 更好的方式
function addLoadEvent(func){
    var oldOnload = window.onload;	//把window.onload事件处理函数的值存入变量ondowload
    if (typeof window.onload != 'function'){//如果window.onload事件没有绑定任何事件处理函数，则为
				//其添加新的事件处理函数
        window.onload = func;
    }else{	//如果window.onload事件已绑定事件处理函数，
				//则重新绑定事件处理函数，在事件处理函数中
				//先执行原来已绑定的事件处理函数，然后调用新添加的事件处理函数
        window.onload = function(){
            oldOnload();
            func();
        }
    }
} 

addLoadEvent(function(){

});
addLoadEvent(function(){
    alert("二次调用load事件");
});

});
```
